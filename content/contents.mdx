import { graphql } from 'gatsby'

export const pageQuery = graphql`
  query Contents {
    allMdx {
      nodes {
        fields {
          created
          updated
        }
        frontmatter {
          title
        }
        slug
      }
    }
  }
`

<ul style={{ listStyle: 'none', paddingLeft: 0 }}>
    {props.data.allMdx.nodes.map(({ fields, frontmatter, slug }) => (
        <li key={slug}>
            <a href={`/${slug}`}>{frontmatter.title}</a>
        </li>
    ))}
</ul>
